📐 Slide 5: Sintaks Generic Dasar

Type parameter: <T>, <E>, <K, V>
Konvensi: T = Type, E = Element, K = Key, V = Value, R = Return type

class MyClass<T> {
  T data;
  MyClass(this.data);
  T getData() => data;
}

T myFunc<T>(T value) => value;

void main() {
  var myIntClass = MyClass<int>(10);
  var myStringClass = MyClass<String>("test");
  print(myIntClass.getData());   // 10
  print(myStringClass.getData());// test
}

📦 Slide 6: Kelas Box<T> Praktis

class Box<T> {
  T? _value;
  Box(this._value);
  T? get value => _value;
  void setValue(T value) { _value = value; }
}

void main() {
  var intBox = Box<int>(10);
  print('Int Box Value: ${intBox.value}');    // 10
  intBox.setValue(20);
  print('After set: ${intBox.value}');        // 20

  var stringBox = Box<String>("Hello, Generic!");
  print('String Box: ${stringBox.value}');
  stringBox.setValue('Hello World');
  print('Updated: ${stringBox.value}');
}

🔗 Slide 7: Multiple Type Parameters (Pair)

class Pair<K, V> {
  K key;
  V value;
  Pair(this.key, this.value);
  void printPair() => print('Key: $key, Value: $value');
}

void main() {
  var pair1 = Pair<String, int>('age', 30);
  pair1.printPair();  // Key: age, Value: 30
  var pair2 = Pair<String, String>('name', 'Alice');
  pair2.printPair();  // Key: name, Value: Alice
  var pair3 = Pair<int, double>(1, 2.71);
  pair3.printPair();  // Key: 1, Value: 2.71
}

⚙️ Slide 8: Fungsi Generic combine<T,U>

String combine<T, U>(T value1, U value2) {
  return '$value1, $value2';
}

void main() {
  var result1 = combine<String, int>("Number is ", 10);
  print(result1); // Number is , 10
  var result2 = combine<int, double>(10, 20.5);
  print(result2); // 10, 20.5
}

✨ Menggabungkan dua nilai dengan tipe berbeda secara type-safe.

🧠 Slide 10: Type Inference (Dart cerdas)

T myFunc<T>(T value) => value;

void main() {
  var intValue = myFunc(10);         // Dart infer T = int
  print(intValue.runtimeType);       // int
  var stringValue = myFunc("test");  // T = String
  print(stringValue.runtimeType);    // String
}

📌 Slide 11: Manfaat & Pembatasan Tipe (Bounds)

  • 🔒 Keamanan Tipe Lebih Baik — mengurangi runtime error.
  • 📖 Peningkatan Keterbacaan — ekspektasi tipe eksplisit.
  • ⚡ Memungkinkan Method Spesifik — akses method dari batasan tipe.
  • 🚫 Mencegah Tipe Tidak Sesuai — penyalahgunaan tipe.
  • 📐 Desain Terstruktur — implementasi generic yang lebih solid.
void printIterable<T extends Iterable<dynamic>>(T iterable) {
  for (var item in iterable) print(item);
}
void main() {
  printIterable<List<int>>([1,2,3]);   // 1,2,3
  printIterable<Set<String>>({'apple', 'banana'});
  // printIterable<int>(10); // Error: int bukan Iterable
}

🗄️ Contoh Implementasi: Cache<T> Generic

class Cache<T> {
  final Map<String, T> _storage = {};
  void set(String key, T value) => _storage[key] = value;
  T? get(String key) => _storage[key];
  void remove(String key) => _storage.remove(key);
}

void main() {
  var userCache = Cache<String>();
  userCache.set('name', 'Alice');
  var productCache = Cache<Product>(); // asumsi class Product ada
  productCache.set('p1', Product('Laptop', 1000));
}

Generic memudahkan pembuatan penyimpanan dengan tipe terjamin.

  final Map<int, T> _store = {};
  void save(int id, T item) { _store[id] = item; }

}

🔄 Contoh Lanjutan: swap<T> & Triple<A,B,C>

void swap<T>(List<T> list, int i, int j) {
  final temp = list[i];
  list[i] = list[j];
  list[j] = temp;
}

class Triple<A, B, C> {
  A first;
  B second;
  C third;
  Triple(this.first, this.second, this.third);
  @override
  String toString() => 'Triple($first, $second, $third)';
}

void main() {
  var numbers = [1, 2, 3];
  swap(numbers, 0, 2);
  print(numbers); // [3, 2, 1]

  var triple = Triple<int, String, bool>(100, "Dart Generics", true);
  print(triple); // Triple(100, Dart Generics, true)
}

🎯 Generic memungkinkan kode yang elegan, fleksibel, tanpa duplikasi.

📚 Best Practice Generic di Dart

  • Gunakan type parameter yang bermakna (T, E, K, V) sesuai konvensi.
  • Batasi tipe dengan extends jika perlu memanggil method tertentu.
  • Manfaatkan type inference untuk menulis kode lebih ringkas.
  • Hindari dynamic jika bisa menggunakan generic demi keamanan compile-time.
  • Gunakan generic pada collection: List<int>, Map<String, User>.
